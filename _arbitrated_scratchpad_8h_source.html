110  void compute_bank_request(req_t &curr_cli_req, bank_req_t bank_req[NumInputs],
111  bank_sel_t bank_sel[NumInputs],
112  bool bank_req_valid[NumInputs]) {
113 
114  // For each request, compute the bank request fields
115  #pragma hls_unroll yes
116  for (unsigned in_chan = 0; in_chan < NumInputs; in_chan++) {
117 
118  // Get the target bank
119  bank_sel[in_chan] =
120  nvhls::get_slc<log2_nbanks>(curr_cli_req.addr[in_chan], 0);
121 
122  // Compile the bank request
123  bank_req[in_chan].do_store = (curr_cli_req.valids[in_chan] == true) &&
124  (curr_cli_req.type.val == CLITYPE_T::STORE);
125  bank_req[in_chan].addr = nvhls::get_slc<addr_width - log2_nbanks>(
126  curr_cli_req.addr[in_chan], log2_nbanks);
127  if (bank_req[in_chan].do_store) {
128  bank_req[in_chan].wdata = curr_cli_req.data[in_chan];
129  }
130 
131  // Cast current input to input index type
132  input_sel_t input_idx = in_chan;
133  bank_req[in_chan].input_chan = input_idx;
134 
135  // Set the request valid flag
136  bank_req_valid[in_chan] = (curr_cli_req.valids[in_chan] == true);
137  }

143  #pragma hls_unroll yes

145  if (bank_req_valid[bank] == true) {
