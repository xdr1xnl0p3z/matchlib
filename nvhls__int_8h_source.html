634 template <unsigned int W1, typename type1, typename type2>
635 inline type2 leading_ones(type1 X) {
636  const unsigned int W2 = log2_ceil<W1>::val;
637  enum { P2 = next_pow2<(W1 + 1) / 2>::val };
638 
639  typename nvhls_t<W1>::nvuint_t X_temp = X;
640  enum { Wu = (W1 > P2) ? (W1 - P2) : 1 };
641  typename nvhls_t<Wu>::nvuint_t upper = get_slc<Wu>(X_temp, P2);
642  typename nvhls_t<P2>::nvuint_t lower = get_slc<P2>(X_temp, 0);
643  typename nvhls_t<W2>::nvuint_t idx = 0;
644 
645  enum { logWu = (log2_ceil<Wu>::val > 0) ? log2_ceil<Wu>::val : 1 };
646  enum { logP2 = (log2_ceil<P2>::val > 0) ? log2_ceil<P2>::val : 1 };
647  typename nvhls_t<logWu>::nvuint_t idxu = 0;
648  typename nvhls_t<P2>::nvuint_t idxl = 0;
649  if (upper != 0) {
650  idxu = leading_ones<Wu, typename nvhls_t<Wu>::nvuint_t,
651  typename nvhls_t<logWu>::nvuint_t>(upper);
652  idx = idxu | P2;
653  } else {
654  idxl = leading_ones<P2, typename nvhls_t<P2>::nvuint_t,
655  typename nvhls_t<logP2>::nvuint_t>(lower);
656  idx = idxl;
657  }
658  return idx;
659 }

700 template <typename type1>
701 unsigned int lzd(type1 X) {
702  unsigned int l;
703 #ifdef HLS_CATAPULT
704  if (X!=0) {
705  l = X.leading_sign();
706  } else {
707  l = Wrapped<type1>::width -1;
708  }
709  return l;
710 #else
711 #if defined(DPOPT)
712  HLS_DPOPT_REGION(DPOPT_DEFAULT, "leading_ones", "leading_ones");
713  HLS_CONSTRAIN_REGION(0, 1);
714 #endif
715  const unsigned int W1 = Wrapped<type1>::width;
716  const unsigned int W2 = log2_ceil<W1>::val;
717  typedef typename nvhls_t<W2>::nvuint_t type2;
718  const bool is_signed = Wrapped<type1>::is_signed;
719  l = leading_ones<W1, type1, type2>(X);

922 bool normalize(type1& X, type2& Y) {
923  bool b;
924 #ifdef HLS_CATAPULT
925  b = X.normalize(Y);
926  return b;
927 #else
928  const unsigned int W1 = Wrapped<type1>::width;
929  const unsigned int W2 = Wrapped<type2>::width;
930  const bool type2_signed = Wrapped<type2>::is_signed;
931  const unsigned int logW1 = log2_ceil<W1>::val;
932  typedef typename nvhls_t<logW1>::nvuint_t lzd_type;
933  typedef typename nvhls_t<logW1 + 1>::nvint_t signed_lzd_type;
934  lzd_type l = lzd(X);
935  lzd_type shift;
936  type2 min_Y = get_min_val<type2>();
937  // If W>logW1, then (Y-l) can be stored in type2 if it is signed or
938  // signed_type2 if type2 is unsigned
939  if (W2 > logW1) {
940  if (type2_signed) {
941  type2 Y_l = Y - l;
942  if (Y_l < min_Y) {
943  shift = Y - min_Y;
944  Y = min_Y;
945  b = false;
946  } else {
947  shift = l;
948  Y = Y_l;
949  b = true;
950  }
951  } else {
952  typedef typename nvhls_t<W2 + 1>::nvint_t signed_type2;
953  signed_type2 Y_l = Y - l;
954  if (Y_l < ((signed_type2)min_Y)) {
955  shift = Y - min_Y;
956  Y = min_Y;
957  b = false;
958  } else {
959  shift = l;
960  Y = Y_l;
961  b = true;
962  }
963  }
964  } else {
965  signed_lzd_type Y_l = Y - l;
966  if (Y_l < ((signed_lzd_type)min_Y)) {
967  shift = Y - min_Y;
968  Y = min_Y;
969  b = false;
970  } else {
971  shift = l;
972  Y = Y_l;
973  b = true;
974  }
975  }
976  if (X == (type1)0) {
977  Y = min_Y;
978  b = true;
979  }
980  X = left_shift(X, shift);
981  return b;
982 #endif
